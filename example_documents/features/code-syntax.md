```typescript
const helloWorld = () => {
  console.log('Hello World!')
}

function blah() {
  hellowWorld();
}

class Dog {
  member = 'm';
  
  init() {
    console.log('initialize');
  }
}
```

```python
def factorial(n):
    if n == 0:
        return 1
    else:
        return n * factorial(n-1)

print(factorial(5))
```

```javascript
// JavaScript example
function greet(name) {
  console.log(`Hello, ${name}!`);
}

greet('World');
```
